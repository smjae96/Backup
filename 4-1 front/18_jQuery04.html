* 데이터 통신 (Ajax)

- JavaScript 방식

        // XMLHttpRequest 객체 사용
        // Ajax 요청 초기화
        const 변수명 = new XMLHttpRequest();
        변수명.open(요청방식, 요청주소);

        // 요청 상태 변화 추적
        변수명.onreadystatechange = function() {
            // readyState : 4 (완료)
            // 요청 상태가 완료가 아닌 경우 메소드 종료
            if(변수명.readyState != 4) return;  

            // 요청에 대한 성공 여부 => status (HTTP status) : 200
            if(변수명.status == 200) {
                // 요청 성공 시 처리할 내용
                // responseText : 요청 결과 
                console.log(변수명.responseText);
            } else {
                // 요청 실패 시 처리할 내용
                // 4xx: 요청 오류(클라이언트)
                // 5xx: 응답 오류(서버)
                // console.log(변수명.status); 
            }
        }

- jQuery 방식

        $.ajax({
            url: 요청주소[,
            method: 요청방식,
            data: 요청 시 전달할 데이터
            ],
            success: function(변수명) {
                // 요청 성공 시 처리할 내용
                // 변수명: 요청 결과를 담은 변수
            },
            error: function(변수명) {
                // 요청 실패 시 처리할 내용
                // 변수명: 요청 실패에 대한 에러 내용을 담은 변수
            }
        })
